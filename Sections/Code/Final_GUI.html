            set(app.StartButton, 'Enable', 'off');

            pause(1); % Remove Later

            % Run if Final Configuration Valid
            t1 = {app.UITable_2.Data{:,2}};

            position = {app.UITable.Data{:,1}};
            colors_data_cell = {app.UITable.Data{:,2}};

            position = cell2mat(position);
            colors_data_array = [colors.red, colors.yellow, colors.red, colors.blank, ...
                 colors.red, colors.yellow, colors.red, colors.blank];
            for i = 1:length(colors_data_cell)
                if(colors_data_cell(i) == "red")
                    colors_data_array(i) = colors.red;
                elseif(colors_data_cell(i) == "yellow")
                    colors_data_array(i) = colors.yellow;
                elseif(colors_data_cell(i) == "green")
                    colors_data_array(i) = colors.green;
                elseif(colors_data_cell(i) == " ")
                    colors_data_array(i) = colors.blank;
                end
            end

%             if(count(cell2mat(t1), 'yellow') ~= count(cell2mat(t2), 'yellow') || count(cell2mat(t1), 'green') ~= count(cell2mat(t2), 'green') || count(cell2mat(t1), 'red') ~= count(cell2mat(t2), 'red'))

These washers can be placed at any location.'}, 'Error');
%             else
                % Start Configuring
                T1 = clock;
                TIME = etime(T1,T1);
                app.TimeOutput.Value = 'Running';

                     colors.blank colors.blank colors.blank, colors.blank];
%                 three = [colors.yellow colors.red colors.red, colors.blank];

                w_sort(position, two, colors_data_array);

                %set_param(Location{1}, 'Value', '0');

                time_elapsed = etime(clock,T1);
                float_time = TIME + time_elapsed;
                % Format the Time String
                secs = abs(float_time);
                s = sprintf('%1.3f',secs);

                    s = sprintf('0%1.3f',secs);
                end

                app.TimeOutput.Value = s;

   %         end
            set(app.StartButton, 'Enable', 'on');

        % Button pushed function: Read Board Button
        function ReadBoard_Callback(app, event)
            % Disable Buttons
            set(app.StartButton, 'Enable', 'off');
            set(app.ReadBoardButton, 'Enable', 'off');

            pause(2);
            % Get Colors and Positions: gameState Structure

            % Update Initial Configuration
            % Empty Data

%             background_subtraction;
            %data = {};

            % The data must have standard angles.

            % Below is the way to Update Initial Configuration
            % This is only for ONE washer, so this will need to be done for
            % EACH washer

            %data = {data; gameState.Angle, gameState.Color};

            % Update Initial Configuration
            %app.UITable_2.Data = data;

            data = {0 ' '; 45 ' '; 90 ' '; 135 ' '; 180 ' '; 225 ' '; 275 ' '; 315 ' '};
            background_subtraction;
            for k = 1:items(1)
                if (gamestate.angle{k} == 0)
                    data{1,1} = gamestate.angle{k};
                    data{1,2} = gamestate.color{k};
                elseif (gamestate.angle{k} == 45)
                    data{2,1} = gamestate.angle{k};
                    data{2,2} = gamestate.color{k};
                elseif (gamestate.angle{k} == 90)
                    data{3,1} = gamestate.angle{k};
                    data{3,2} = gamestate.color{k};
                elseif (gamestate.angle{k} == 135)
                    data{4,1} = gamestate.angle{k};
                    data{4,2} = gamestate.color{k};
                elseif (gamestate.angle{k} == 180)
                    data{5,1} = gamestate.angle{k};
                    data{5,2} = gamestate.color{k};
                elseif (gamestate.angle{k} == 225)
                    data{6,1} = gamestate.angle{k};
                    data{6,2} = gamestate.color{k};
                elseif (gamestate.angle{k} == 275)
                    data{7,1} = gamestate.angle{k};
                    data{7,2} = gamestate.color{k};
                elseif (gamestate.angle{k} == 315)
                    data{8,1} = gamestate.angle{k};
                    data{8,2} = gamestate.color{k};
                end
            end

            current_colors = [colors.red, colors.yellow, colors.red, colors.blank, ...
                colors.red, colors.yellow, colors.red, colors.blank];

            for i = 1:length(colors_data_cell)
                if(gamestate.color{k} == "red")
                    current_colors(i) = colors.red;
                elseif(gamestate.color{k} == "yellow")
                    current_colors(i) = colors.yellow;
                elseif(gamestate.color{k} == "green")
                    current_colors(i) = colors.green;
                elseif(gamestate.color{k} == " ")
                    current_colors(i) = colors.blank;
                end
            end

            % Default until above is implemented
            %data = {30 ' '; 60 ' '; 90 ' '; 135 ' '; 180 ' '; -45 ' '; -90 ' '; -120 ' '; -150 ' '};
            %data = {gamestate.angle(0) gamestate.color(0)};
            %gamestate.angle = {0 1 2 3 4};
%             a = gamestate.color.';
%             b = gamestate.angle.';
                app.UITable_2.Data(1:8,:) = data;
%             app.UITable_2.Data(1:length(gamestate.color),2) = a;
%             app.UITable_2.Data(1:length(gamestate.color),1) = b;
            %app.UITable_2.Data = data;

            % Re-Enable Buttons
            set(app.StartButton, 'Enable', 'on');
            set(app.ReadBoardButton, 'Enable', 'on');
        end
    end

    % Component initialization
    methods (Access = private)

        % Create UIFigure and components
        function createComponents(app)

            % Create UIFigure and hide until all components are created
            app.UIFigure = uifigure('Visible', 'off');
            app.UIFigure.Position = [100 100 640 535];
            app.UIFigure.Name = 'UI Figure';

            % Create StartButton
            app.StartButton = uibutton(app.UIFigure, 'push');
            app.StartButton.BackgroundColor = [0 1 0];
            app.StartButton.FontName = 'Bookman';
            app.StartButton.FontColor = [1 1 1];
            app.StartButton.Position = [400 400 100 34];
            app.StartButton.Text = 'Start';
            app.StartButton.ButtonPushedFcn = createCallbackFcn(app, @Start_Callback, true);

            % Create ReadBoardButton_2
            app.ReadBoardButton = uibutton(app.UIFigure, 'push');
            app.ReadBoardButton.BackgroundColor = [1 0 0];
            app.ReadBoardButton.FontName = 'Bookman';
            app.ReadBoardButton.FontColor = [1 1 1];
            app.ReadBoardButton.Position = [100 400 100 34];
            app.ReadBoardButton.Text = 'Read Board';
            app.ReadBoardButton.ButtonPushedFcn = createCallbackFcn(app, @ReadBoard_Callback, true);

            % Create FinalConfigurationPanel
            app.FinalConfigurationPanel = uipanel(app.UIFigure);
            app.FinalConfigurationPanel.Title = 'Final Configuration';
            app.FinalConfigurationPanel.FontName = 'Bookman';
            app.FinalConfigurationPanel.Position = [250 150 203 205];


            % Create UITable
            app.UITable = uitable(app.FinalConfigurationPanel);
            app.UITable.ColumnName = {'Location'; 'Color'};
            app.UITable.RowName = {''};
            app.UITable.ColumnEditable = [false true];
            app.UITable.FontName = 'Bookman';
            app.UITable.Position = [0 -63 204 250];
            app.UITable.RowName = {};
            app.UITable.ColumnFormat = ({[] {' ' 'red' 'yellow' 'green'}});
            data = {0 ' '; 45 ' '; 90 ' '; 135 ' '; 180 ' '; 225 ' '; 275 ' '; 315 ' '};
            app.UITable.Data = data;



            % Create InitialConfigurationPanel
            app.InitialConfigurationPanel = uipanel(app.UIFigure);
            app.InitialConfigurationPanel.Title = 'Initial Configuration';
            app.InitialConfigurationPanel.FontName = 'Bookman';
            app.InitialConfigurationPanel.Position = [30 150 203 205];

            % Create UITable_2 - Subject to Change
            app.UITable_2 = uitable('Parent', app.InitialConfigurationPanel);
            app.UITable_2.ColumnName = {'Location'; 'Color'};
            app.UITable_2.RowName = {};
            app.UITable_2.ColumnEditable = [false true];
            app.UITable_2.FontName = 'Bookman';
            app.UITable_2.Position = [0 -63 204 250];
            app.UITable_2.ColumnFormat = ({[] []});

            %data = {30 ' '; 60 ' '; 90 ' '; 135 ' '; 180 ' '; -45 ' '; -90 ' '; -120 ' '; -150 ' '};
            % Use if cannot read board
            data = {0 ' '; 45 ' '; 90 ' '; 135 ' '; 180 ' '; 225 ' '; 275 ' '; 315 ' '};

            %data = {30 'Yellow'; 60 'Red'; 90 'Green'; 135 'Green'; 180 ' '; -45 ' '; -90 ' '; -120 ' '; -150 ' '};
            app.UITable_2.Data = data;

            % Create SolveTimesPanel
            app.SolveTimesPanel = uipanel(app.UIFigure);
            app.SolveTimesPanel.Title = 'Solve Time (seconds)';
            app.SolveTimesPanel.FontName = 'Bookman';
            app.SolveTimesPanel.Position = [470 150 136 44];

            % Create TimeOutput
            app.TimeOutput = uieditfield(app.SolveTimesPanel);
            app.TimeOutput.FontName = 'Bookman';
            app.TimeOutput.Position = [0 1 135 22];
            app.TimeOutput.Editable = false;

            % Show the figure after all components are created
            app.UIFigure.Visible = 'on';
        end
    end

    % App creation and deletion
    methods (Access = public)

        % Construct app
        function app = Final_GUI

            % Create UIFigure and components
            createComponents(app)

            % Register the app with App Designer
            registerApp(app, app.UIFigure)

            if nargout == 0
                clear app
            end
        end

        % Code that executes before app deletion
        function delete(app)

            % Delete UIFigure when app is deleted
            delete(app.UIFigure)
        end

    end
end
